    def Zobrist.hash(matrix)

      # should seed...
      r ||= Array.new(3) { |i| Array.new(3) { |j| Array.new(3) { |k| rand } } }

      key = 0
       matrix.each_with_index do |row, x|
        row.each_with_index do |value, y|
          key ^= r[x][y][value]
        end
      end
      key
    end
