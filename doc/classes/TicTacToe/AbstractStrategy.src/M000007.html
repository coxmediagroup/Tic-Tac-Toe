    def best_move(state, player=1, debug=false)

      best_val = -INFINITY
      best_position = nil
      pp state if debug
      each_immediate_successor_state_with_index(state) do |succ, i|
        move_value = value(succ, player, 0) # , self)

        puts "=== got move value #{move_value} for move #{i}" if debug

        if move_value >= best_val
          best_val = move_value
          best_position = i
        end
      end

      puts "=== returning #{best_position} as best position" if debug

      # return best position
      best_position

    end
