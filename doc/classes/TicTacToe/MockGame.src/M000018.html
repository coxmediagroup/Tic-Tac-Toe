    def play_immediate_successors(state, debug=true)
      scores = [0,0,0]
      each_immediate_successor_state(state) do |successor|
        scores[play(successor, debug)] += 1
      end
      puts "==== draws: #{scores[0]} / wins: #{scores[1]} / losses: #{scores[2]} ===" if debug
      scores
    end
