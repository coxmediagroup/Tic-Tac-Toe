    def pretty(state,value_map=[' ','X','O'])
      txt = StringIO.new

      m = matrix(state)
      m.each_with_index do |row, n|
        txt << row.map do |value|
          mark = value_map[value]
          line = " #{mark} "
          line
        end.join('|') << "\n"

        # draw a line between each row
        txt << "-"*((3*3)+(2)) << "\n" unless n == 2
      end

      txt.rewind
      txt.read
    end
